\newcommand{\tab}[1]{
	% weils sonst anscheinend keinen tab gibt in latex
	% verwenden:
	% \tab{1} \tab{2}
	% \tab{3} \tab{4}
	% gibt:
	% 1    2
	% 3    4
	\settowidth{\tabcont}{#1}
	\ifthenelse{\lengthtest{\tabcont < .25\linewidth}}
	{\makebox[.25\linewidth][l]{#1}\ignorespaces}
	{\makebox[.5\linewidth][l]{#1}\ignorespaces}
}
